```typescript
'use client'

import { useState, useEffect, useCallback } from 'react'
import { useQueryStates } from 'nuqs'
import { taskSearchParams } from '@/app/(app)/tasks/search-params'
import { TaskSearchBar } from './TaskSearchBar'
import { TaskFilters } from './TaskFilters'
import { TaskCard } from './TaskCard'
import { EmptyState, NoResultsIcon, NoTasksIcon } from './EmptyState'
import {
  searchTasksAction,
  filterTasksAction,
  getTagsForFilterAction,
} from '@/app/actions/search'
import { Task, Tag } from '@prisma/client'
import Link from 'next/link'

type TaskWithTags = Task & { tags: Tag[] }

type TaskListWithFiltersProps = {
  initialQuery: string
  initialStatuses: ('TODO' | 'IN_PROGRESS' | 'DONE')[]
  initialTags: string[]
  sortBy: 'createdAt' | 'updatedAt' | 'title'
  sortOrder: 'asc' | 'desc'
}

type GroupedTasks = {
  TODO: TaskWithTags[]
  IN_PROGRESS: TaskWithTags[]
  DONE: TaskWithTags[]
}

function groupTasksByStatus(tasks: TaskWithTags[]): GroupedTasks {
  return {
    TODO: tasks.filter(t => t.status === 'TODO'),
    IN_PROGRESS: tasks.filter(t => t.status === 'IN_PROGRESS'),
    DONE: tasks.filter(t => t.status === 'DONE'),
  }
}

function TaskSection({
  title,
  tasks,
  searchQuery,
}: {
  title: string
  tasks: TaskWithTags[]
  searchQuery?: string
}) {
  if (tasks.length === 0) return null

  return (
    <div>
      <h2 className="text-lg font-semibold text-gray-900 dark:text-gray-100 mb-4">
        {title} ({tasks.length})
      </h2>
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {tasks.map((task) => (
          <TaskCard key={task.id} task={task} />
        ))}
      </div>
    </div>
  )
}

export function TaskListWithFilters({
  initialQuery,
  initialStatuses,
  initialTags,
  sortBy,
  sortOrder,
}: TaskListWithFiltersProps) {
  const [filters, setFilters] = useQueryStates(taskSearchParams)
  const [tasks, setTasks] = useState<TaskWithTags[]>([])
  const [availableTags, setAvailableTags] = useState<Tag[]>([])
  const [isLoading, setIsLoading] = useState(true)
  const [isSearching, setIsSearching] = useState(false)
  const [resultCount, setResultCount] = useState<number>()

  // Load initial data and tags
  useEffect(() => {
    async function loadInitialData() {
      setIsLoading(true)
      try {
        // Load available tags
        const tagsResult = await getTagsForFilterAction()
        if (tagsResult.success) {
          setAvailableTags(tagsResult.tags)
        }

        // Load tasks based on initial filters
        await loadTasks()
      } finally {
        setIsLoading(false)
      }
    }

    loadInitialData()
  }, []) // eslint-disable-line react-hooks/exhaustive-deps

  // Load tasks when filters change
  const loadTasks = useCallback(async () => {
    const hasSearch = filters.q.trim().length > 0
    const hasFilters = filters.status.length > 0 || filters.tags.length > 0

    if (hasSearch) {
      // Use search action
      setIsSearching(true)
      const result = await searchTasksAction(filters.q)
      if (result.success) {
        let filteredTasks = result.tasks

        // Apply additional filters on search results
        if (filters.status.length > 0) {
          filteredTasks = filteredTasks.filter(t =>
            filters.status.includes(t.status as 'TODO' | 'IN_PROGRESS' | 'DONE')
          )
        }
        if (filters.tags.length > 0) {
          filteredTasks = filteredTasks.filter(t =>
            t.tags.some(tag => filters.tags.includes(tag.id))
          )
        }

        setTasks(filteredTasks)
        setResultCount(filteredTasks.length)
      }
      setIsSearching(false)
    } else {
      // Use filter action
      const result = await filterTasksAction({
        status: filters.status.length > 0 ? filters.status : undefined,
        tagIds: filters.tags.length > 0 ? filters.tags : undefined,
      })
      if (result.success) {
        setTasks(result.tasks)
        setResultCount(hasFilters ? result.tasks.length : undefined)
      }
    }
  }, [filters.q, filters.status, filters.tags])

  // Reload when filters change
  useEffect(() => {
    loadTasks()
  }, [loadTasks])

  const handleSearch = useCallback(async (query: string) => {
    // URL is already updated by TaskSearchBar via setFilters
    // loadTasks will be triggered by the useEffect above
  }, [])

  const handleClearFilters = useCallback(() => {
    setFilters({
      q: '',
      status: [],
      tags: [],
      sortBy: 'createdAt',
      sortOrder: 'desc',
    })
  }, [setFilters])

  const hasActiveFilters = filters.q || filters.status.length > 0 || filters.tags.length > 0

  // Group tasks by status for display
  const groupedTasks = groupTasksByStatus(tasks)
  const totalTasks = tasks.length
  const hasAnyTasks = totalTasks > 0

  if (isLoading) {
    return (
      <div className="animate-pulse space-y-8">
        {/* Search skeleton */}
        <div className="h-10 bg-gray-200 dark:bg-gray-700 rounded-lg w-full max-w-md" />

        {/* Task sections skeleton */}
        {[1, 2, 3].map((i) => (
          <div key={i}>
            <div className="h-6 bg-gray-200 dark:bg-gray-700 rounded w-32 mb-4" />
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
              {[1, 2].map((j) => (
                <div
                  key={j}
                  className="h-32 bg-gray-200 dark:bg-gray-700 rounded-lg"
                />
              ))}
            </div>
          </div>
        ))}
      </div>
    )
  }

  return (
    <div className="flex gap-6">
      {/* Filter sidebar */}
      <aside className="w-64 flex-shrink-0 hidden lg:block">
        <div className="sticky top-4 space-y-6">
          <TaskFilters
            selectedStatuses={filters.status}
            selectedTags={filters.tags}
            availableTags={availableTags}
            onStatusChange={(statuses) => setFilters({ status: statuses })}
            onTagChange={(tagIds) => setFilters({ tags: tagIds })}
            onClearFilters={handleClearFilters}
          />
        </div>
      </aside>

      {/* Main content */}
      <div className="flex-1 min-w-0 space-y-6">
        {/* Search bar */}
        <TaskSearchBar
          value={filters.q}
          onChange={(q) => setFilters({ q })}
          onSearch={handleSearch}
          isSearching={isSearching}
          resultCount={hasActiveFilters ? resultCount : undefined}
        />

        {/* Mobile filter toggle (shown on small screens) */}
        <div className="lg:hidden">
          <details className="border border-gray-200 dark:border-gray-700 rounded-lg">
            <summary className="px-4 py-2 cursor-pointer text-sm font-medium text-gray-700 dark:text-gray-300">
              Filters {hasActiveFilters && `(${filters.status.length + filters.tags.length} active)`}
            </summary>
            <div className="p-4 border-t border-gray-200 dark:border-gray-700">
              <TaskFilters
                selectedStatuses={filters.status}
                selectedTags={filters.tags}
                availableTags={availableTags}
                onStatusChange={(statuses) => setFilters({ status: statuses })}
                onTagChange={(tagIds) => setFilters({ tags: tagIds })}
                onClearFilters={handleClearFilters}
              />
            </div>
          </details>
        </div>

        {/* Results */}
        {!hasAnyTasks ? (
          hasActiveFilters ? (
            <EmptyState
              icon={<NoResultsIcon />}
              title="No tasks found"
              description="Try adjusting your search or filters to find what you're looking for"
              action={
                <button
                  onClick={handleClearFilters}
                  className="
                    px-4 py-2
                    bg-blue-600 text-white
                    rounded-md font-medium
                    hover:bg-blue-700
                  "
                >
                  Clear all filters
                </button>
              }
            />
          ) : (
            <EmptyState
              icon={<NoTasksIcon />}
              title="No tasks yet"
              description="Create your first task to get started organizing your work"
              action={
                <Link
                  href="/tasks/new"
                  className="
                    px-4 py-2
                    bg-blue-600 text-white
                    rounded-md font-medium
                    hover:bg-blue-700
                  "
                >
                  Create your first task
                </Link>
              }
            />
          )
        ) : (
          <div className="space-y-8">
            <TaskSection title="To Do" tasks={groupedTasks.TODO} searchQuery={filters.q} />
            <TaskSection title="In Progress" tasks={groupedTasks.IN_PROGRESS} searchQuery={filters.q} />
            <TaskSection title="Done" tasks={groupedTasks.DONE} searchQuery={filters.q} />
          </div>
        )}
      </div>
    </div>
  )
}
```

```typescript
type PageProps = {
  searchParams: Promise<Record<string, string | string[] | undefined>>
}

export default async function TasksPage({ searchParams }: PageProps) {
  // Parse search params on server using nuqs cache
  const params = await searchParams
  const { q, status, tags, sortBy, sortOrder } = searchParamsCache.parse(params)

  return (
    <div>
      {/* Page header */}
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-3xl font-bold text-gray-900 dark:text-white">
          My Tasks
        </h1>
        <Link
          href="/tasks/new"
          className="
            inline-flex items-center justify-center
            px-4 py-2
            bg-blue-600 text-white
            rounded-md font-medium
            hover:bg-blue-700
            focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500
            dark:bg-blue-500 dark:hover:bg-blue-600
            transition-colors
          "
        >
          <svg className="w-5 h-5 mr-2" fill="none" viewBox="0 0 24 24" stroke="currentColor">
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 4v16m8-8H4" />
          </svg>
          New Task
        </Link>
      </div>

      {/* Task list with search and filters */}
      <Suspense fallback={<TaskPageSkeleton />}>
        <TaskListWithFilters
          initialQuery={q}
          initialStatuses={status}
          initialTags={tags}
          sortBy={sortBy}
          sortOrder={sortOrder}
        />
      </Suspense>
    </div>
  )
}
```
